200 void WsApplication::setTemplate(const std::string& sPageOrig)
201 {
202  std::string sPage = sPageOrig;
203  boost::algorithm::replace_all(sPage, "&amp;", "&");
204  if ( WsLayoutProperties::instance()->get("global", "use_template", "true") == "true" ) {
205  std::string truePath(WsModules().pathWithoutPrefix(sPage));
206  NodePtr rootNode = m_pUser->getAccessRoot();
207  if ( !rootNode.get() ) {
208  wApp->log("ALERT") << "WsApplication::setTemplate() root node is null !! ";
209  // TODO : ERROR Page
210  wApp->quit();
211  return;
212  }
213  NodePtr curNode = rootNode.get()->eatPath(truePath);
214  if ( !curNode.get() ) {
215  boost::filesystem::path p(truePath);
216  truePath = p.parent_path().string();
217  curNode = rootNode.get()->eatPath(truePath);
218  }
219  // Boucle sur parent, jusqu'a obtention du template
220  std::string sTemplate;
221  std::string sPath;
222  NodePtr pNode = curNode;
223  while (pNode.get()) {
224  sTemplate = pNode.get()->getProperties().get()->get("global", "template", "");
225  if ( sTemplate.size() > 0 ) break;
226  pNode = pNode.get()->getParent();
227  }
228  if ( sTemplate.size() < 1 ) sTemplate = "root.tpl"; // ce template doit exister
229  if ( sPage.compare(0, 5, "/Edit" ) == 0 ) sTemplate = "stdEditPage.tpl"; // cui-ci aussi
230  if ( sPage.compare(0, 8, "/SiteMap" ) == 0 ) sTemplate = "stdContact.tpl"; // cui-ci aussi
231  if ( sPage.compare(0, 5, "/Logo" ) == 0 ) sTemplate = "GhFullWidth.tpl"; // cui-ci aussi
232  if ( sPage.compare(0, 7, "/Search" ) == 0 ) sTemplate = "stdContact.tpl"; // cui-ci aussi
233  if ( pNode.get() ) {
234  wApp->log("notice") << "WsApplication::setTemplate() - Using template : " << sTemplate << " with home page : " << sPage;
235  sPath = sPage;
236  } else {
237  wApp->log("notice") << "WsApplication::setTemplate() - Using template : " << sTemplate << ", with home page : " << m_sHomePage;
238  sPath = m_sHomePage;
239  }
240  if ( sTemplate != m_sTemplate ) {
241  m_sTemplate = sTemplate;
242  root()->clear();
243  m_pButtonsBar = 0;
244  m_pContent = 0; // le template doit prevoir la creation du content
245  m_pImages = 0;
246  WsTemplate* pTemplate = new WsTemplate(sTemplate, sPath);
247  bool bUseLayout = false; // TODO : Put this option in the layout.json
248  if ( bUseLayout ) {
249  WVBoxLayout* vbox = new WVBoxLayout();
250  vbox->setContentsMargins(0, 0, 0, 0);
251  vbox->addWidget(pTemplate, 1);
252  root()->setLayout(vbox);
253  } else {
254  root()->addWidget(pTemplate);
255  // root()->resize(WLength(100, WLength::Percentage), WLength(100, WLength::Percentage));
256  }
257  return;
258  }
259  }
260  if ( m_pContent )
261  m_pContent->setPath(sPage);
262 }
