78 int WsFsTreeModification::createNode( const std::set<std::string>& groups, const string& uid, const string& p, int type)
79 {
80  NodePtr n;
81  boost::filesystem::path parent = p;
82  boost::filesystem::path root = m_updater->getLastTree()->getRootPath();
83  parent = parent.parent_path();
84  FileSystemTreePtr ft = m_updater->getLastTree();
85  /* Test if parent directory exists */
86  n = ft->eatPath(parent.string());
87  if ( n.get() == 0) {
88  LOG(ERROR) << "WsFsTreeClient::createNode() : Parent Node not found " << parent.string();
89  return FAILURE;
90  }
91  /* Test if parent is a dirNode */
92  if ( !n.get()->isDirectory()) {
93  LOG(ERROR) << "WsFsTreeClient::createNode() : Parent Node is not a directory " << parent.string();
94  return FAILURE;
95  }
96  /* If not allowed and is not editor nor admin return failure */
97  /* FIXME editor has right to edit something he doesen't have access to ? */
98  if (!n.get()->isAllowed(groups) && (!isEditor(groups) || !isAdministrator(groups) )) {
99  LOG(ERROR) << "WsFsTreeClient::createNode() : User " << uid << " not allowed to edit " << parent.string();
100  return FAILURE;
101  }
102  /* Check if new node already exit */
103  if ( boost::filesystem::exists(root / p) ) {
104  LOG(INFO) << "WsFsTreeClient::createNode() : Node already exist" << p;
105  return FAILURE;
106  }
107  switch (type) {
108  case 0: { /* Create a FileNode */
109  fstream f;
110  /* Create the file on disk */
111  f.open( string(root.string() + p).c_str(), ios::out );
112  if (!f.is_open())
113  return FAILURE;
114  f.close();
115  /* Create Node and properties */
116  WsFileNode* fn = new WsFileNode(root / p, root);
117  NodePropertiesPtr props(new WsNodeProperties(root / path(p), WsNodeProperties::File));
118  fn->setProperties(props);
119  NodePtr nn = NodePtr(fn);
120  /* Insert Node */
121  m_updater->getLastTree()->insertNode(nn);
122  return SUCCESS;
123  }
124  case 1: { /* DirNode */
125  /* Create Node */
126  try {
127  /* Create the directory on disk */
128  boost::filesystem::create_directory(root / path(p));
129  } catch (std::exception& e) {
130  LOG(ERROR) << "WsFsTreeClient::createNode() : could not create node " << e.what();
131  return FAILURE;
132  }
133  /* Create Config Files */
134  try {
135  boost::filesystem::create_directory(root / path(p) / GlobalConfig::PathToConfig);
136  NodePropertiesPtr fp(new WsNodeProperties( root / path(p), WsNodeProperties::Dir));
137  set<string> groups;
138  groups.insert(uid);
139  fp->setGroups(groups);
140  fp->save();
141  WsDirNode* dn = new WsDirNode(root / p, root);
142  dn->setProperties(fp);
143  NodePtr nn = NodePtr(dn);
144  /* Insert the new Node */
145  m_updater->getLastTree()->insertNode(nn);
146  return SUCCESS;
147  } catch (std::exception& e) {
148  LOG(ERROR) << "WsFsTreeClient::createNode() : could not create node " << e.what();
149  return FAILURE;
150  }
151  }
152  default:
153  return FAILURE;
154  }
155 }
