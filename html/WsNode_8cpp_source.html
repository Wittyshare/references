205 NodePtr WsNode::eatPath(const string& path)
206 {
207  string name;
208  string sub;
209  //We are searching for this node
210  if (path == "/") {
211  NodePtr sThis = shared_from_this();
212  return sThis;
213  }
214  //remove trailing '/'
215  if (path[0] == '/')
216  name = path.substr(1);
217  else name = path;
218  //get the next chunck of the name
219  if (name.find("/") != string::npos) {
220  sub = name.substr(0, name.find_first_of("/"));
221  name = name.substr(name.find_first_of("/"));
222  } else {
223  sub = name;
224  name = "";
225  }
226  NodePtr cur = this->getNodeByName(sub);
227  if (cur.get() == 0) {

229  return NodePtr();
230  }
231  if (name == "" || name == "/") {
