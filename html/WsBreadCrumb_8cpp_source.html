57 void WsBreadCrumb::doPathChanged(std::string newPathOrig)
58 {

60  boost::algorithm::replace_all(newPath, "&amp;", "&");

70  std::string sPath = "/";
71  if ( m_bDebug )

73  boost::filesystem::path myPath(sWithoutPrefix);
74  if ( myPath.string() != sPath )
75  while ( myPath.has_parent_path() ) {
76  breadCrump bc2;

78  if ( curNode ) bc2.title = curNode.get()->getDisplayName(true);

80  bc2.path = myPath.string();
81  bc2.node = curNode;
82  m_vBreadCrumb.insert(m_vBreadCrumb.begin() + 1, bc2);
83  myPath = myPath.parent_path();
84  if ( myPath.string() == sPath ) break;
85  }
86  Wt::WStackedWidget* contents = new Wt::WStackedWidget();
87  m_pMenu = new Wt::WMenu(contents, Wt::Horizontal);
88  m_pMenu->itemSelected().connect(SLOT(this, WsBreadCrumb::onBreadCrumbSelected));
89  addWidget(m_pMenu);
