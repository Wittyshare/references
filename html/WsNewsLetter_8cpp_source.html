125 WsNewsLetter::emailAvailability WsNewsLetter::isEmailExistInLdap(const std::string& email)
126 {
127  gdCLdapServer m_cLdapServer;
128  m_cLdapServer.strServer = asString(option("ldapServer")).narrow();
129  m_cLdapServer.strDisplayName = asString(option("ldapName")).toUTF8();
130  m_cLdapServer.strLogon = asString(option("ldapLogon")).toUTF8();
131  m_cLdapServer.strPassword = asString(option("ldapSecret")).toUTF8();
132  if ( m_cLdapServer.TryServer() )
133  WApplication::instance()->log("notice") << " WsNewsLetter::isEmailExistInLdap() Ldap try ok !";
134  else {
135  std::string error = "WsNewsLetter::isEmailExistInLdap() Ldap try not ok !";
136  WApplication::instance()->log("notice") << error;
137  wApp->setInternalPath("/ws/Errors/ServerTempUnavailable.fhtml", true);
138  return ldapError;
139  }
140  gdCLdapAttributes cWitchAttrs;
141  cWitchAttrs.push_back(new gdCLdapAttribute("givenName"));
142  cWitchAttrs.push_back(new gdCLdapAttribute("sn"));
143  cWitchAttrs.push_back(new gdCLdapAttribute("mail"));
144  cWitchAttrs.push_back(new gdCLdapAttribute("dn"));
145  gdCLdap m_cLdap;
146  m_cLdap.setDebugging(true);
147  m_cLdap.setManageDSAIT(true);
148  gdCLdapFilter cFilter;
149  cFilter.pServer = &m_cLdapServer;
150  cFilter.strInitialSearchBase = asString(option("ldapBase")).toUTF8();
151  cFilter.nScope = LDAP_SCOPE_SUB;
152  cFilter.strFilter = "(&(mail=" + email + "))";
153  gdCLdapEntries cEntries;
154  if ( !m_cLdap.GetInfo(cFilter, cWitchAttrs, cEntries) && m_cLdap.GetLastError() != CLdapErrNoEntries ) {
155  std::string notice = "WsNewsLetter::isEmailExistInLdap() cannot get ldap info or no entries!";
156  WApplication::instance()->log("notice") << notice;
157  wApp->setInternalPath("/ws/Errors/ServerTempUnavailable.fhtml", true);
158  return ldapError;
159  }
160  // Email not available
161  if ( cEntries.size() == 0 ) {
162  emailNotInLdap(email);
163  return emailNotAvailable;
164  }
165  // Multiple email entries
166  if ( cEntries.size() > 1 ) {
167  multipleEmail(email);
168  return emailMultiple;
169  }
170  std::string sGivenName;
171  std::string sSN;
172  std::string sEmail;
173  std::string sDn;
174  for (int nAttr = 0; nAttr < cEntries[0]->Attrs.size(); ++nAttr) {
175  if ( cEntries[0]->Attrs[nAttr]->Attr == "sn" ) sSN = cEntries[0]->Attrs[nAttr]->Values[0]->ToStr();
176  if ( cEntries[0]->Attrs[nAttr]->Attr == "givenName" ) sGivenName = cEntries[0]->Attrs[nAttr]->Values[0]->ToStr();
177  if ( cEntries[0]->Attrs[nAttr]->Attr == "mail" ) sEmail = cEntries[0]->Attrs[nAttr]->Values[0]->ToStr();
178  if ( cEntries[0]->Attrs[nAttr]->Attr == "dn" ) sDn = cEntries[0]->Attrs[nAttr]->Values[0]->ToStr();
179  }
180  std::string notice = "WsNewsLetter::isEmailExistInLdap() match givenName, " + sGivenName + " sn, " + sSN + " email, " + sEmail + " dn, " + sDn;
181  WApplication::instance()->log("notice") << notice;
182  gdCLdapEntry entry;
183  entry.Dn = sDn;
184  gdCLdapAttribute updateAttr("EurListsMember");
185  updateAttr.bNewAttribute = true; // false destroy all existing values, true add a new one, in all case if the attribute does not exist it is added
186  updateAttr.addValue("Newsletter03");
187  // if the value exist : we receive error 312 "Type of value exists"
188  if ( !m_cLdapServer.UpdateAttribute(entry, updateAttr) )
189  wApp->log("notice") << "WsNewsLetter::isEmailExistInLdap() Cannot update ldap entry : " << sDn << "Contain a correct value";
190  else
191  wApp->log("notice") << "WsNewsLetter::isEmailExistInLdap() Ldap entry : " << sDn << " correctly updated !";
192 }

236  vmime::ref<vmime::net::session> sess = vmime::create<vmime::net::session>();
237  vmime::utility::url url(sSmtpServer);

271  std::string text = "Email is already in db, nothing to do !";

278  eToken->token = hash;

283  }
284  }
285  if ( bMatchEmail ) {
286  wApp->setInternalPath("/ws/Errors/AlreadyInDb.fhtml", true);
287  return;
