187 int WsModZip::handleFiles(int t)
188 {
189  LOG(DEBUG) << "WsModZip :: Handling only files";
190  /* Get the WsUser class */
191  WsUser* user = WsApplication::wsInstance()->wsUser();
192  NodePtr accessTree = user->getAccessRoot();
193  if (!accessTree.get()) {
194  LOG(ERROR) << "WsModZip::handleFiles() AccessRoot is NULL";
195  return FAILURE;
196  }
197  /* Get the internalPath */
198  string path = WApplication::instance()->internalPath();
199  path = WsApplication::wsInstance()->WsModules().pathWithoutPrefix(path);
200  LOG(DEBUG) << "WsModZip :: Archiving " << path;
201  /* Get the directory contents */
202  NodePtr tmp = accessTree.get()->eatPath(path);
203  if (!tmp)
204  return FAILURE;
205  vector<NodePtr> contents = tmp.get()->getAll();
206  LOG(DEBUG) << "WsModZip :: Dir contents size :" << contents.size();
207  vector<NodePtr>::iterator it;
208  int size = contents.size();
209  int i = 0;
210  /* Iterate over all files and folders in directory */
211  for (it = contents.begin(); it != contents.end(); ++it) {
212  /* If it's a file */
213  if ((*it).get()->isRegularFile()) {
214  LOG(DEBUG) << "WsModZip :: Typeid is file for " << (*it)->getFullPath().string();
215  /* Check if we can include it in zip (read config) */
216  if ((*it).get()->getProperties().get()->get(WSMODZIP, ALLOW_ZIP, "true") == "true")
217  if ((*it).get()->getSize() > 0)
218  /* If yes and we file size is not 0 add it to zip */
219  appendToZip((*it).get()->getPath().string());
220  }
221  }
222  return writeAndCloseZip(t);
223 }

308 int WsModZip::writeAndCloseZip(int format)
309 {
310  LOG(DEBUG) << "WsModZip :: Generating archive file with " << m_paths.size() << " files";
311  char buff[16384];
312  struct archive* a;
313  struct archive* disk;
314  struct archive_entry* entry;
315  ssize_t len;
316  int fd;
317  WsApplication* wsApp = WsApplication::wsInstance();
318  WsUser* user = wsApp->wsUser();
319  char* currentDir = get_current_dir_name();
320  string rootPath = user->getRootPath();
321  const char* temp = rootPath.c_str();
322  char* newDir = new char[rootPath.size() + 1]();
323  LOG(DEBUG) << "WsModZip :: Root path is " << rootPath;
324  strncpy(newDir, temp, rootPath.size());
325  newDir[rootPath.size()] = '\0';
326  /* Do a chdir to zip files correctly */
327  LOG(DEBUG) << "WsModZip :: Changing current dir name from " << currentDir << " to " << newDir;
328  if (chdir(newDir) != 0) {
329  LOG(ERROR) << "WsModZip :: Could not chdir to " << newDir;
330  return -1;
331  }
332  LOG(DEBUG) << "WsModZip :: Current dir name : " << get_current_dir_name();
333  string ext;
334  /* Create a new archive */
335  a = archive_write_new();
336  /* Check format wanted */
337  if (format == TAR) {
338  LOG(DEBUG) << "WsModZip :: Archive format will be tar.bz2";
339  /* Set flags TAR */
340  archive_write_set_compression_compress(a);
341  archive_write_set_compression_bzip2(a);
342  archive_write_set_format_ustar(a);
343  ext = ".tar.bz2";
344  } else if (format == ZIP) {
345  /* Set flags zIP */
346  archive_write_set_compression_compress(a);
347  LOG(DEBUG) << "WsModZip :: Archive format will be zip";
348  archive_write_set_format_zip(a);
349  ext = ".zip";
350  } else {
351  LOG(ERROR) << "WsModZip :: Unknown file format ";
352  return -1;
353  }
354  /* Get TMP directory of user */
355  string filenameTmp = wsApp->getUserTmpPath() + string(basename(tmpnam(NULL))) + ext;
356  LOG(DEBUG) << "WsModZip :: Archive file name will be " << filenameTmp.c_str();
357  /* Open archive */
358  archive_write_open_file(a, filenameTmp.c_str());
359  disk = archive_read_disk_new();
360  archive_read_disk_set_standard_lookup(disk);
361  /* Nothing to archive */
362  if (m_paths.size() == 0) {
363  m_errmsg = "Nothing to archive";
364  return -1;
365  }
366  /* Iterate over paths we want to add to the archive */
367  for (int i = 0; i < m_paths.size(); ++i) {
368  struct archive* disk = archive_read_disk_new();
369  int r;
370  LOG(DEBUG) << "WsModZip :: Adding file " << m_paths[i];
371  string str = m_paths[i];
372  /* remove the starting / */
373  if (m_paths[i][0] == '/') {
374  str = m_paths[i].substr(1);
375  LOG(DEBUG) << "WsModZip :: m_paths starts with /";
376  }
377  LOG(DEBUG) << "Reading file " << str.c_str();
378  r = archive_read_disk_open(disk, str.c_str());
379  if (r != ARCHIVE_OK) {
380  LOG(ERROR) << "WsModZip :: Cannot read file";
381  return -1;
382  }
383  /* Write */
384  for (;;) {
385  int needcr = 0;
386  entry = archive_entry_new();
387  r = archive_read_next_header2(disk, entry);
388  if (r == ARCHIVE_EOF)
389  break;
390  if (r != ARCHIVE_OK) {
391  LOG(ERROR) << "WsModZip :: Archive error";
392  return -1;
393  }
394  archive_read_disk_descend(disk);
395  LOG(DEBUG) << "WsModZip :: Writing archive header";
396  r = archive_write_header(a, entry);
397  if (r < ARCHIVE_OK) {
398  needcr = 1;
399  }
400  if (r == ARCHIVE_FATAL) {
401  LOG(ERROR) << "WsModZip :: Archive fatal";
402  return -1;
403  }
404  if (r > ARCHIVE_FAILED) {
405  fd = open(archive_entry_sourcepath(entry), O_RDONLY);
406  len = read(fd, buff, sizeof(buff));
407  while (len > 0) {
408  LOG(DEBUG) << "WsModZip :: writing data of length " << sizeof(buff);
409  archive_write_data(a, buff, len);
410  len = read(fd, buff, sizeof(buff));
411  }
412  close(fd);
413  } else
414  LOG(ERROR) << "WsModZip :: Archive failed";
415  archive_entry_free(entry);
416  }
417  archive_read_close(disk);
418  archive_read_free(disk);
419  }
420  archive_write_close(a);
421  archive_write_free(a);
422  if (chdir(currentDir) != 0)
423  LOG(DEBUG) << "WsModZip :: Could not chdir to " << currentDir;
424  LOG(DEBUG) << "WsModZip :: Archive ready for Download";
425  boost::replace_all(filenameTmp, wsApp->docRoot(), "");
426  WApplication::instance()->redirect(filenameTmp);
427  m_dialog->hide();
428  delete m_dialog;
429  return 0;
