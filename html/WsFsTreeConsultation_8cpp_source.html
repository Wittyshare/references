27 int WsFsTreeConsultation::getPermissions( const std::set<std::string>& groups, const std::string& p)
28 {
29  NodePtr n;
30  FileSystemTreePtr ft = m_updater->getLastTree();
31  /* Get node */
32  n = ft->eatPath(p);
33  if (n.get() == 0) {
34  /* Path not found */
35  LOG(INFO) << "WsFsTreeConsultation::getPermissions() : No such file or directory :" << p;
36  return GlobalConfig::NotFound;
37  }
38  if (groups.count(m_conf->get("global", "admin_group", "administrator")) > 0) {
39  /* user is admin , readwrite perms*/
40  return GlobalConfig::ReadWrite;
41  }
42  WsGlobalProperties* props = WsGlobalProperties::instance();
43  if (props->get("global", "public_site", "false") == "true") {
44  /* Public site read access granted */
45  return GlobalConfig::Read;
46  }
47  /* Check if user is allowed */
48  if (n.get()->isAllowed(groups)) {
49  if (groups.count(m_conf->get("global", "editor_group", "editor")) > 0) {
50  /* user is editor. read write perms */
51  return GlobalConfig::ReadWrite;
52  } else {
53  /* user is a normal user. read perms */
54  return GlobalConfig::Read;
55  }
56  } else {
57  /* no access */
58  return GlobalConfig::NoAccess;
59  }
60  return FAILURE;
61 }
