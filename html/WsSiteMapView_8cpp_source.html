295 bool WsSiteMapView::loadTree(WStandardItem* currentItem, NodePtr pNode, gdWFileView::skipMode skipParentDir)
296 {
297  WsUser* pUser = WsApp->wsUser();
298  if (!pNode.get())
299  return false;
300  if ( m_bLogSiteMap )
301  if ( currentItem->model() == m_pModelTree )
302  if ( m_bLogSiteMap )
303  wApp->log("notice") << "WsSiteMapView::loadTree() m_pModelTree pNode full path = " << pNode.get()->getPath().string();
304  else if ( m_bLogSiteMap )
305  wApp->log("notice") << "WsSiteMapView::loadTree() m_pModelView pNode full path = " << pNode.get()->getPath().string();
306  if ( pNode.get()->isDirectory() ) {
307  if ( !skipParentDir && (pNode != m_pTreecurNode) && !(m_tbFlags & gdWFileView::tbUseLeftTree) ) {
308  std::vector<WStandardItem*> pRow;
309  std::string newPath = pNode.get()->getPath().string();
310  WsSiteMapItem* newItem = new WsSiteMapItem("..");
311  newItem->setData("1" + path2SiteMapPath(newPath), UserRole);
312  newItem->setIcon(wApp->theme()->resourcesUrl() + "gdwtcore/Icons/folder.png");
313  pRow.push_back(newItem);
314  pRow.push_back(new WsSiteMapItem());
315  pRow.push_back(new WsSiteMapItem());
316  currentItem->appendRow(pRow);
317  }
318  std::vector<NodePtr> tmpNode;
319  std::vector<NodePtr>& dirNode = tmpNode;
320  if ( currentItem->model() == m_pModelTree ) dirNode = pNode.get()->getDirectories();
321  else dirNode = pNode.get()->getAll();
322  if ( m_bLogSiteMap )
323  wApp->log("notice") << "WsSiteMapView::loadTree() dirNode size = " << dirNode.size();
324  for (std::vector<NodePtr>::iterator it = dirNode.begin(); it != dirNode.end(); ++it) {
325  NodePtr curNode = *it;
326  // Items not displayed in view are still visible in the sitemap to the administrator
327  if ( !(pUser->isAdministrator() || pUser->isEditor()) )
328  if ( curNode.get()->getProperties().get()->get("global", "in_view", "true") != "true" ) continue;
329  // Do not display ws.res directories in the sitemap
330  if (curNode.get()->getName() == "ws.res") continue;
331  std::string curPath = m_sRootPath + curNode.get()->getPath().string();
332  if ( m_bLogSiteMap )
333  wApp->log("notice") << "WsSiteMapView::loadTree() curPath = " << curPath;
334  if ( curNode.get()->isDirectory() ) {
335  if ( m_bLogSiteMap )
336  wApp->log("notice") << "WsSiteMapView::loadTree() detected directory";
337  if ( !loadFolder(currentItem, curPath, curNode) ) return false;
338  } else {
339  if ( m_bLogSiteMap )
340  wApp->log("notice") << "WsSiteMapView::loadTree() detected file";
341  if ( currentItem->model() == m_pModelTree ) continue;
342  if ( !loadFile(currentItem, curPath, curNode) ) return false;
343  }
344  }
345  if ( currentItem->model() == m_pModelView )
346  m_pModelView->sort(0);
347  }
348  return true;
349 }
