74 bool WsNodeProperties::isAllowed(std::set<string> gids)
75 {
76  std::set<string> nodeGroups = getGroups();
77  if (!m_parsed) {
78  string p = getPath();
79  parse(p);
80  }
81  WsGlobalProperties* props = WsGlobalProperties::instance();
82  string admgrp = props->get("global", "admin_group", "");
83  /* Admin has access to everything */
84  if (gids.count(admgrp) > 0 ) {
85  LOG(DEBUG) << "WsNodeProperties::isAllowed() : User is Admin. Access allowed for " << getPath();
86  return true;
87  }
88  std::set<string>::iterator itGids;
89  /* Iterate over the list of users allowed and check */
90  /* if there is a similar entry in the gids list */
91  for (itGids = gids.begin(); itGids != gids.end(); ++itGids) {
92  if (nodeGroups.count(*itGids) > 0) {
93  /* We found a similar entry, we need to check publish date */
94  /* Editor has access to node even if not published yet */
95  string edtgrp = props->get("global", "editor_group", "");
96  if (gids.count(edtgrp) > 0 ) {
97  LOG(DEBUG) << "WsNodeProperties::isAllowed() User is Editor. Access allowed for " << getPath();
98  return true;
99  }
100  LOG(DEBUG) << "WsNodeProperties::isAllowed() : User not editor";
101  try {
102  time_t pnode = boost::lexical_cast<time_t>(this->get("global", "publish_date", "0"));
103  time_t pnow = time(NULL);
104  if (pnode <= pnow) {
105  return true;
106  } else {
107  LOG(DEBUG) << "WsNodeProperties::isAllowed() : File is not published yet. Access Denied for " << getPath();
108  return false;
109  }
110  } catch (boost::bad_lexical_cast&) {
111  LOG(ERROR) << "WsNodeProperties::isAllowed() : Cannot parse publish date for " << this->getPath();
112  return false;
113  }
114  }
115  }
116  /* No similar entries found. Group has no access */
117  LOG(DEBUG) << "WsNodeProperties :: User has no access ";
118  return false;
119 }
